# Unge UI

ungeUI 是一个 Vue3 组件库

# 安装

yarn add ungeui

# 使用

import { createApp } from 'vue'
import UngeUI from 'ungeui'
import 'ungeui/style/index.css'

const app = createApp(App)
app.use(UngeUI)
